function myIsNaN(value) {
  return typeof value === 'number' && value !== value;
}

// Пример:
let testValue = 'abc';
if (myIsNaN(testValue)) {
  console.log('Это не число или NaN.');
} else {
  console.log('Это число.');
}

function checkProbabilityTheory(count) {
  let evenCount = 0;
  let oddCount = 0;

  for (let i = 0; i < count; i++) {
    const randomNumber = Math.floor(Math.random() * (1000 - 100 + 1) + 100);

    if (randomNumber % 2 === 0) {
      evenCount++;
    } else {
      oddCount++;
    }
  }

  const totalNumbers = evenCount + oddCount;
  const evenPercentage = (evenCount / totalNumbers) * 100;
  const oddPercentage = (oddCount / totalNumbers) * 100;

  console.log(`Количество сгенерованых чисел: ${totalNumbers}`);
  console.log(`Парных чисел: ${evenCount}`);
  console.log(`Не парных чисел: ${oddCount}`);
  console.log(`Процент парных до не парных: ${evenPercentage.toFixed(2)}% : ${oddPercentage.toFixed(2)}%`);
}

// Пример:
checkProbabilityTheory(1000);
